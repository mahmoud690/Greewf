	در کنسول پکیج منجر ، دقت کنید که پکیج سورس بر روی همه ست شده باشد

scaffold Controller 'AienTehran.PIN.Biz.Entities.xxx' -DbContextType 'AienTehran.PIN.Dal.PINContext'  -Repository 

just controller:
scaffold Controller 'AienTehran.PIN.Biz.Entities.xxx' -DbContextType 'AienTehran.PIN.Dal.PINContext' -NoChildItems 
scaffold Controller 'AienTehran.PIN.Biz.Entities.xxx' -DbContextType 'AienTehran.PIN.Dal.PINContext' -NoChildItems -Repository 

just repository:
	scaffold Repository 'AienTehran.PIN.Biz.Entities.xxx' -DbContextType 'AienTehran.PIN.Dal.PINContext'
